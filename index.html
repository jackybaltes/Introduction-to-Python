def maxOdd(lst):
  "returns max. odd number in a list"
  m = None
  for e in lst:
    if ( e % 2 == 1 ):
      if m is None or ( e > m ):
        m = e
  return m

  "returns max. number that follows immediately an odd number in a list"
  m = None
  for i in range(len(lst)-1):
    e = lst[i]
    if e % 2 == 1:
